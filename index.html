tableBody.appendChild(row); }); } // Função para submeter um novo demon completion document.querySelector('#completion-form').addEventListener('submit', function(event) { event.preventDefault(); const nome = document.querySelector('#name').value; demonList.push(newDemon); // Adiciona o novo demon renderDemonList(); // Re-renderiza a lista updateLeaderboard(newDemon); // Atualiza o leaderboard document.querySelector('#completion-form').reset(); // Reseta o formulário }); // Função para atualizar o leaderboard function updateLeaderboard(demon) { const leaderboardList = document.querySelector('#leaderboard-list'); const newEntry = document.createElement('li');
